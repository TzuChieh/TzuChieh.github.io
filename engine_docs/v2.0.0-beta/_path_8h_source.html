44  // Constructing a path from some string representing of the path. The string
45  // can be either a relative or absolute path, or even a path fragment. The
46  // constructed path substitutes all separators to a system specific (preferred)
47  // one.
48  //
49  inline explicit Path(const std::string& path) :
50  m_path(std_filesystem::path(path).make_preferred())
51  {}
52 
53  inline bool isRelative() const
54  {

56  }

63  inline std::string toAbsoluteString() const
64  {
65  const std::string& absPath = std_filesystem::absolute(m_path).string();
66  if(!Path(absPath).isAbsolute())
67  {
68  std::cerr << "warning: at Path::getAbsoluteString(), "
69  << "path <"
70  << m_path.string()
71  << "> cannot convert to absolute path" << std::endl;
72  }
73 
74  return absPath;
75  }
76 
77  // Appending one path to another. System specific directory separators are
78  // added in between two path objects.
79  //
80  inline Path append(const Path& other) const
81  {
82  auto thisPath = this->removeTrailingSeparator();
83  auto otherPath = other.removeLeadingSeparator();
84  return Path((thisPath.m_path /= otherPath.m_path).string());
85  }

87  inline std::string toString() const

90  }

92  inline Path removeLeadingSeparator() const
93  {
94  std::string pathStr = m_path.string();
95  while(!pathStr.empty())
96  {
97  if(charToWchar(pathStr.front()) == m_path.preferred_separator)
98  {
99  pathStr.erase(0, 1);
100  }
101  else
102  {
103  break;
104  }
105  }
106 
107  return Path(pathStr);
108  }
109 
110  inline Path removeTrailingSeparator() const
111  {
112  std::string pathStr = m_path.string();
113  while(!pathStr.empty())
114  {
115  if(charToWchar(pathStr.back()) == m_path.preferred_separator)
116  {
117  pathStr.pop_back();
118  }
119  else
120  {
121  break;
122  }
123  }
124 
125  return Path(pathStr);
126  }
