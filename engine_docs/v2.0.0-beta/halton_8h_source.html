19 template<uint64 BASE>
20 inline real radical_inverse(const uint64 value)
21 {
22  static_assert(BASE >= 2);
23 
24  if constexpr(BASE == 2)
25  {
26  return math::reverse_bits(value) * 0x1p-64_r;
27  }
28  else
29  {
30  constexpr real rcpBase = 1.0_r / BASE;
31 
32  uint64 currentValue = value;
33  uint64 unscaledInversedValue = 0;
34  real scaler = 1.0_r;
35  while(currentValue > 0)
36  {
37  const uint64 quotient = currentValue / BASE;
38  const uint64 remainder = currentValue - quotient * BASE;
39 
40  unscaledInversedValue = unscaledInversedValue * BASE + remainder;
41  scaler *= rcpBase;
42 
43  currentValue = quotient;
44  }

50 template<uint64 BASE, typename P>
51 inline real radical_inverse_permuted(const uint64 value, const P* const permutation)
52 {
53  static_assert(BASE >= 2);
54 
55  PH_ASSERT(permutation);
56 
57  constexpr real rcpBase = 1.0_r / BASE;
58 
59  uint64 currentValue = value;
60  uint64 unscaledInversedValue = 0;
61  real scaler = 1.0_r;
62  while(currentValue > 0)
63  {
64  const uint64 quotient = currentValue / BASE;
65  const uint64 remainder = currentValue - quotient * BASE;
66  const uint64 permutedRemainder = permutation[remainder];
67 
68  unscaledInversedValue = unscaledInversedValue * BASE + permutedRemainder;
69  scaler *= rcpBase;
70 
71  currentValue = quotient;
72  }
73 
74  // The permutation table may map the digit (remainder) 0 to a nonzero
75  // value. The while loop cannot catch this case since the iteration stops
76  // prematurely once <currentValue> becomes 0, missing an infinitely long
77  // suffix of digits with value <permutation[0]>. Fortunately, as stated in
78  // the book of pbrt-v3 this is a geometric series with a simple analytic
79  // solution whose value can be added here.
80  // Note that <permutation[0] * rcpBase> is the first term of the geometric
81  // series and <rcpBase> is the common ratio.
82  return std::clamp(
83  (unscaledInversedValue + permutation[0] * rcpBase / (1.0_r - rcpBase)) * scaler,
84  0.0_r, 1.0_r);
85 }
