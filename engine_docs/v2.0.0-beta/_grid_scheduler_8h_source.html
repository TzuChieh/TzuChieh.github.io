104 inline void GridScheduler::scheduleOne(WorkUnit* const out_workUnit)
105 {
106  PH_ASSERT(out_workUnit);
107 
108  // Cell coordinates are always in the canonical Cartesian space. Mapping
109  // only performed on divided ranges.
110 
111  if(m_currentCell.x() < m_numCells.x() && m_currentCell.y() < m_numCells.y())
112  {
113  const std::size_t totalWidth = m_totalWorkUnit.getRegion().getWidth();
114  const std::size_t totalHeight = m_totalWorkUnit.getRegion().getHeight();
115 
116  auto sideRangeX = math::ith_evenly_divided_range(
117  m_currentCell.x(), totalWidth, m_numCells.x());
118  if(m_origin == EOrigin::LOWER_RIGHT || m_origin == EOrigin::UPPER_RIGHT)
119  {
120  const auto size = sideRangeX.second - sideRangeX.first;
121  sideRangeX.first = totalWidth - sideRangeX.second;
122  sideRangeX.second = sideRangeX.first + size;
123  }
124 
125  auto sideRangeY = math::ith_evenly_divided_range(
126  m_currentCell.y(), totalHeight, m_numCells.y());
127  if(m_origin == EOrigin::UPPER_LEFT || m_origin == EOrigin::UPPER_RIGHT)
128  {
129  const auto size = sideRangeY.second - sideRangeY.first;
130  sideRangeY.first = totalHeight - sideRangeY.second;
131  sideRangeY.second = sideRangeY.first + size;
132  }
133 
134  *out_workUnit = WorkUnit(
135  Region(
136  math::TVector2<int64>(math::Vector2S(sideRangeX.first, sideRangeY.first)) + m_totalWorkUnit.getRegion().getMinVertex(),
137  math::TVector2<int64>(math::Vector2S(sideRangeX.second, sideRangeY.second)) + m_totalWorkUnit.getRegion().getMinVertex()),
138  m_totalWorkUnit.getDepth());
139 
140  if(m_prioriAxis == math::constant::X_AXIS)
141  {
142  ++m_currentCell.x();

144  {
