67 inline bool SidednessAgreement::isSidednessAgreed(
68  const SurfaceHit& X,
69  const math::Vector3R& vec) const
70 {
71  switch(m_policy)
72  {
73  case ESidednessPolicy::DO_NOT_CARE:
74  case ESidednessPolicy::TRUST_GEOMETRY:
75  case ESidednessPolicy::TRUST_SHADING:
76  {
77  // no agreement issue with single input vector with these policies
78  return true;
79  break;
80  }
81 
82  case ESidednessPolicy::STRICT:
83  {
84  const math::Vector3R& Ng = X.getGeometryNormal();
85  const math::Vector3R& Ns = X.getShadingNormal();
86 
87  return Ng.dot(vec) * Ns.dot(vec) > 0.0_r;
88  break;
89  }
90 
91  default:
92  PH_ASSERT_UNREACHABLE_SECTION();
93  return false;
94  break;
95  }
