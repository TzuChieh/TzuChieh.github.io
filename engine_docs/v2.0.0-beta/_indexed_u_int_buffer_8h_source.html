116 inline void IndexedUIntBuffer::setUInt(const std::size_t index, const IntegerType value)
117 {
118  PH_ASSERT(isAllocated());
119 
120  const auto uint64Value = lossless_integer_cast<uint64>(value);
121  if(uint64Value > getMaxAllowedValue())
122  {
123  throw std::invalid_argument(std::format(
124  "Integer value {} cannot be stored using {} bits.", value, m_numBitsPerUInt));
125  }
126 
127  const std::size_t firstByteIndex = index * m_numBitsPerUInt / CHAR_BIT;
128  const std::size_t firstByteBitOffset = index * m_numBitsPerUInt - firstByteIndex * CHAR_BIT;
129  const std::size_t numStraddledBytes = (firstByteBitOffset + m_numBitsPerUInt + (CHAR_BIT - 1)) / CHAR_BIT;
130 
131  PH_ASSERT_LT(firstByteBitOffset, CHAR_BIT);
132  PH_ASSERT_LE(numStraddledBytes, 8 + 1);
133  PH_ASSERT_LE(firstByteIndex + numStraddledBytes, m_byteBufferSize);
134 
135  // Potentially read straddled previous & next values' bits
136  uint64 rawBits = 0;
137  std::memcpy(&rawBits, &m_byteBuffer[firstByteIndex], std::min<std::size_t>(numStraddledBytes, 8));
138 
139  // Store value between the back of previous value and the head of next value
140  rawBits = math::clear_bits_in_range(rawBits, firstByteBitOffset, firstByteBitOffset + m_numBitsPerUInt);
141  rawBits |= (uint64Value << firstByteBitOffset);
142  std::memcpy(&m_byteBuffer[firstByteIndex], &rawBits, std::min<std::size_t>(numStraddledBytes, 8));
143 
144  // Handle situations where the value needs the 9-th byte (straddles next byte), this can happen
145  // since we support any number of bits per index
146  if(numStraddledBytes > 8)
147  {
148  uint8 remainingRawBits;
149  std::memcpy(&remainingRawBits, &m_byteBuffer[firstByteIndex + 8], 1);
150 
151  // Store the remaining value before next value

153  remainingRawBits |= static_cast<uint8>(uint64Value >> (64 - firstByteBitOffset));
154  std::memcpy(&m_byteBuffer[firstByteIndex + 8], &remainingRawBits, 1);

191 inline uint64 IndexedUIntBuffer::getUInt(const std::size_t index) const
192 {
193  PH_ASSERT(isAllocated());
194 
195  const std::size_t firstByteIndex = index * m_numBitsPerUInt / CHAR_BIT;
196  const std::size_t firstByteBitOffset = index * m_numBitsPerUInt - firstByteIndex * CHAR_BIT;
197  const std::size_t numStraddledBytes = (firstByteBitOffset + m_numBitsPerUInt + (CHAR_BIT - 1)) / CHAR_BIT;
198 
199  PH_ASSERT_LT(firstByteBitOffset, CHAR_BIT);
200  PH_ASSERT_LE(numStraddledBytes, 8 + 1);
201  PH_ASSERT_LE(firstByteIndex + numStraddledBytes, m_byteBufferSize);
202 
203  // Read current value's bits (first 8 bytes, at most)
204  uint64 rawBits = 0;
205  std::memcpy(&rawBits, &m_byteBuffer[firstByteIndex], std::min<std::size_t>(numStraddledBytes, 8));
206 
207  // Clear previous and next values' bits if any, then get the value
208  const auto bitMask = math::set_bits_in_range<uint64>(0, firstByteBitOffset, firstByteBitOffset + m_numBitsPerUInt);
209  uint64 value = (rawBits & bitMask) >> firstByteBitOffset;
210 
211  // Handle situations where the value needs the 9-th byte (straddles next byte), this can happen
212  // since we support any number of bits per index
213  if(numStraddledBytes > 8)
214  {
215  uint8 remainingRawBits;
216  std::memcpy(&remainingRawBits, &m_byteBuffer[firstByteIndex + 8], 1);
217 
218  // Extract the remaining value and clear next value's bits if any
219  const auto remainingBitsBitMask = math::set_bits_in_range<uint8>(0, static_cast<std::size_t>(0), firstByteBitOffset + m_numBitsPerUInt - 64);
220  remainingRawBits &= remainingBitsBitMask;
221 
222  // Add the remaining bits to the value
223  value |= (static_cast<uint64>(remainingRawBits) << (64 - firstByteBitOffset));
224  }
