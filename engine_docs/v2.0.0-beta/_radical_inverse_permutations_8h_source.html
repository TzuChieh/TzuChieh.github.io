43 {
44  PH_ASSERT_LE(numDims, MAX_DIMENSIONS);
45 
46  // The offset after index <numDims - 1> is the total size of the buffer
47  m_permutations.resize(PERMUTATION_OFFSETS(numDims));

49  // Generate random digit permutation/mapping for each dimension
