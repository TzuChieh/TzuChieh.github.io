79 inline void ByteBufferInputStream::read(const std::size_t numBytes, std::byte* const out_bytes)
80 {
81  PH_ASSERT(out_bytes);
82 
83  if(canRead(numBytes))
84  {
85  PH_ASSERT_LT(m_readHead, numBufferBytes());
86  PH_ASSERT_LE(m_readHead + numBytes, numBufferBytes());
87 
88  std::copy(
89  byteBuffer() + m_readHead,
90  byteBuffer() + m_readHead + numBytes,
91  out_bytes);
92

95  else
96  {
97  throw IOException(std::format(

103 inline void ByteBufferInputStream::readString(std::string* const out_string, const char delimiter)
104 {
105  PH_ASSERT(out_string);
106 
107  if(canRead())
108  {
109  PH_ASSERT_LT(m_readHead, numBufferBytes());
110 
111  const char* const charBufferView = charBuffer();
112 
113  const std::size_t beginIdx = m_readHead;
114  for(; m_readHead < numBufferBytes(); ++m_readHead)
115  {
116  if(charBufferView[m_readHead] == delimiter)
117  {
118  ++m_readHead;
119  break;
120  }
121  }
122  // <m_readHead> is on either one past delimiter or buffer size after the loop
123 
124  PH_ASSERT_LT(beginIdx, m_readHead);
125 
126  out_string->clear();
127 
128  // Ternary for not to include the delimiter in the output string
129  const auto numChars = m_readHead < numBufferBytes() ? m_readHead - 1 - beginIdx : m_readHead - beginIdx;
130  out_string->append(
131  &(charBufferView[beginIdx]),
132  numChars);
133  }
134  else
135  {
136  throw IOException(std::format(
137  "Attempt to read a string with delimiter {} from buffer range [{}, {}).",
138  delimiter, m_readHead, numBufferBytes()));
139  }
