78  m_grid()
79 {
80  PH_ASSERT_MSG(tileSize.product() > 0, tileSize.toString());
81 
82  // since grid scheduler slices work region, we need to round up the
83  // grid size to make each slice with size <tileSize>
84  TVector2<int64> gridSize(
85  math::ceil_div_positive(totalWorkUnit.getWidth(), static_cast<int64>(tileSize.x)),
86  math::ceil_div_positive(totalWorkUnit.getHeight(), static_cast<int64>(tileSize.y)));
87  gridSize.mulLocal(TVector2<int64>(tileSize));
88 
89  const Region gridWorkRegion(
90  totalWorkUnit.getRegion().minVertex,
91  totalWorkUnit.getRegion().minVertex + gridSize);
92 
93  m_grid = GridScheduler(
94  numWorkers,
95  WorkUnit(gridWorkRegion, totalWorkUnit.getDepth()),
96  Vector2S(gridWorkRegion.getExtents()) / tileSize,
97  toGridOrigin(origin),

101 inline void TileScheduler::scheduleOne(WorkUnit* const out_workUnit)
102 {
103  PH_ASSERT(out_workUnit);
104 
105  // Tile coordinates are always in the canonical Cartesian space, mapping
106  // is performed later on.
107 
108  WorkUnit gridWorkUnit;
109  if(m_grid.schedule(&gridWorkUnit))
110  {
111  Region tileWorkRegion = gridWorkUnit.getRegion();
112  tileWorkRegion.intersectWith(m_totalWorkUnit.getRegion());
113 
114  if(tileWorkRegion.isValid())
115  {
116  *out_workUnit = WorkUnit(tileWorkRegion, gridWorkUnit.getDepth());
117  return;
118  }
119  }
120 
121  *out_workUnit = WorkUnit();
122 }
