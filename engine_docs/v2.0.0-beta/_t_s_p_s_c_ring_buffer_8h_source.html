79  inline void endProduce()
80  {
81  PH_ASSERT(isProducerThread());
82 
83  // Increment before flag release--we want this addition to be visible on consumer
84  m_headDistance.fetch_add(1, std::memory_order_relaxed);
85 
86  Item& currentItem = getCurrentProducerItem();
87  currentItem.isBetweenProduceBeginAndEnd = false;
88 
89  // `item` can no longer be modified by producer
90 
91  // Mark that current item is now set and is ready for being consumed
92  const bool hasAlreadySealed = currentItem.isSealedForConsume.test_and_set(std::memory_order_release);
93  PH_ASSERT(!hasAlreadySealed);
94 
95  // Notify the potentially waiting consumer so it can start consuming this item
96  currentItem.isSealedForConsume.notify_one();
97 
98  advanceProduceHead();
99  }

124  inline void endConsume()
125  {
126  PH_ASSERT(isConsumerThread());
127 
128  // Decrement after flag acquire--will never underflow since every consumption happens after a
129  // production with memory effects made visible
130  m_headDistance.fetch_sub(1, std::memory_order_relaxed);
131 
132  Item& currentItem = getCurrentConsumerItem();
133  currentItem.isBetweenConsumeBeginAndEnd = false;
134 
135  // `item` can no longer be modified by consumer
136 
137  // Mark that current item is now consumed and is ready for use by producer again
138  currentItem.isSealedForConsume.clear(std::memory_order_release);
139 
140  // Notify the potentially waiting producer so it can start producing
141  currentItem.isSealedForConsume.notify_one();
142 
143  advanceConsumeHead();
144  }

201  inline std::size_t getHeadDistanceDirectly() const
202  {
203  // Pad produce head to the next cycle first to ensure positive result, then mod it.
204  //
205  // `return (m_produceHead + N - m_consumeHead) % N;`
206  //
207  // Above is the straightforward way of calculating distance, however the result is ambiguous on
208  // determining whether the buffer is empty or full--returning 0 on both conditions.
209 
210  return lossless_integer_cast<std::size_t>(
211  m_headDistance.load(std::memory_order_relaxed));
212  }
