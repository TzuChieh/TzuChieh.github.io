105 inline auto VPMRadianceEvaluator::impl_onPathHitSurface(
106  const std::size_t pathLength,
107  const SurfaceHit& surfaceHit,
108  const SpectralStrength& pathThroughput) -> ViewPathTracingPolicy
109 {
110  // TODO: remove hardcoded max path length
111  constexpr std::size_t MAX_PATH_LENGTH = 5;
112 
113  PH_ASSERT_GE(pathLength, 1);
114 
115  const PrimitiveMetadata* const metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
116  const SurfaceOptics* const surfaceOptics = metadata->getSurface().getOptics();
117 
118  if(metadata->getSurface().getEmitter())
119  {
120  SpectralStrength viewPathRadiance;
121  metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewPathRadiance);
122  m_sampledRadiance.addLocal(pathThroughput * viewPathRadiance);
123  }
124 
125  // FIXME: properly handle delta optics (mixed case)
126 
127  if(pathLength < MAX_PATH_LENGTH && surfaceOptics->getAllPhenomena().hasAtLeastOne({
128  ESurfacePhenomenon::DELTA_REFLECTION,
129  ESurfacePhenomenon::DELTA_TRANSMISSION}))
130  {
131  return ViewPathTracingPolicy().
132  traceBranchedPathFor(SurfacePhenomena({
133  ESurfacePhenomenon::DELTA_REFLECTION,
134  ESurfacePhenomenon::DELTA_TRANSMISSION})).
135  useRussianRoulette(false);
136  }
137 
138  PH_ASSERT_LE(pathLength, MAX_PATH_LENGTH);
139 
140  m_photonCache.clear();
141  m_photonMap->findWithinRange(surfaceHit.getPosition(), m_kernelRadius, m_photonCache);
142 
143  TSurfaceEventDispatcher<ESaPolicy::STRICT> surfaceEvent(m_scene);
144 
145  const Vector3R L = surfaceHit.getIncidentRay().getDirection().mul(-1);
146  const Vector3R Ns = surfaceHit.getShadingNormal();
147  const Vector3R Ng = surfaceHit.getGeometryNormal();
148 
149  BsdfEvaluation bsdfEval;
150  SpectralStrength radiance(0);
151  for(const auto& photon : m_photonCache)
152  {
153  const Vector3R V = photon.get<EPhotonData::FROM_DIR>();
154 
155  bsdfEval.inputs.set(surfaceHit, L, V, ALL_ELEMENTALS, ETransport::IMPORTANCE);
156  if(!surfaceEvent.doBsdfEvaluation(surfaceHit, bsdfEval))
157  {
158  continue;
159  }
160 
161  SpectralStrength throughput(pathThroughput);
162  throughput.mulLocal(bsdfEval.outputs.bsdf);
163  throughput.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
164 
165  radiance.addLocal(throughput * photon.get<EPhotonData::THROUGHPUT_RADIANCE>());
166  }
167 
168  // OPT: cache
169  const real kernelArea = m_kernelRadius * m_kernelRadius * constant::pi<real>;
170  const real radianceMultiplier = 1.0_r / (kernelArea * static_cast<real>(m_numPhotonPaths));
171 
172  radiance.mulLocal(radianceMultiplier);
173 
174  m_sampledRadiance.addLocal(radiance);
175 
176  return ViewPathTracingPolicy().kill();
177 }
