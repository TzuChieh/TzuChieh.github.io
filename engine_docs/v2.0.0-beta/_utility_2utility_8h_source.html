199 template<std::integral DstType, std::integral SrcType>
200 inline DstType lossless_integer_cast(const SrcType src)
201 {
202  using SrcLimits = std::numeric_limits<SrcType>;
203  using DstLimits = std::numeric_limits<DstType>;
204 
205  // Note that the use of `std::cmp_<X>` functions are important as the comparisons
206  // may be signed <-> unsigned comparisons, which may cause signed limits to overflow
207 
208  // TODO: we may need to cast src to some integer first to support char and bool types (they are not supported by cmp functions)
209 
210  constexpr bool mayHavePositiveOverflow = std::cmp_greater(SrcLimits::max(), DstLimits::max());
211  constexpr bool mayHaveNegativeOverflow = std::cmp_less(SrcLimits::lowest(), DstLimits::lowest());
212 
213  if constexpr(mayHavePositiveOverflow)
214  {
215  if(std::cmp_greater(src, DstLimits::max()))
216  {
217  throw_formatted<OverflowException>("cast results in positive overflow: {} exceeds the limit {}",
218  src, DstLimits::max());
219  }
220  }
221 
222  if constexpr(mayHaveNegativeOverflow)
223  {
224  if(std::cmp_less(src, DstLimits::lowest()))
225  {
226  throw_formatted<OverflowException>("cast results in negative overflow: {} exceeds the limit {}",
227  src, DstLimits::lowest());
228  }
229  }
230 
231  // All possible integer overflow scenarios are checked so it is safe to cast now
232  return static_cast<DstType>(src);
233 }

235 template<std::floating_point DstType, std::floating_point SrcType>
236 inline DstType lossless_float_cast(const SrcType src)
237 {
238  // Nothing to do if both types are the same
239  if constexpr(std::is_same_v<SrcType, DstType>)
240  {
241  return src;
242  }
243  // If we are converting to a wider floating-point type, generally it will be lossless
244  else if constexpr(sizeof(DstType) > sizeof(SrcType))
245  {
246  // We need both types to be IEEE-754
247  static_assert(std::numeric_limits<SrcType>::is_iec559);
248  static_assert(std::numeric_limits<DstType>::is_iec559);
249 
250  return static_cast<DstType>(src);
251  }
252  // Otherwise, cast to `DstType` then back to `SrcType` and see if there is any difference
253  else
254  {
255  const auto dst = static_cast<DstType>(src);
256  const auto dstBackToSrc = static_cast<SrcType>(dst);
257  if(src != dstBackToSrc)
258  {
259  throw_formatted<NumericException>("cast results in numeric precision loss: {} -> {}",
260  src, dstBackToSrc);
261  }
262 
263  return dst;
264  }
