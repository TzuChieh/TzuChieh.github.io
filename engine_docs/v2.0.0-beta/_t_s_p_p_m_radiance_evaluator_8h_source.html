111  const Vector2R& filmNdc,
112  const SpectralStrength& pathThroughput)
113 {
114  // FIXME: sample res
115 
116  const real fFilmXPx = filmNdc.x * static_cast<real>(m_film->getActualResPx().x);

118 
119  const Vector2S regionPosPx(Vector2R(

141 template<typename Viewpoint, typename Photon>
142 inline auto TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onPathHitSurface(
143  const std::size_t pathLength,
144  const SurfaceHit& surfaceHit,
145  const SpectralStrength& pathThroughput) -> ViewPathTracingPolicy
146 {
147  const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
148  const SurfaceOptics* optics = metadata->getSurface().getOptics();
149 
150  // TODO: MIS
151  if constexpr(Viewpoint::template has<EViewpointData::VIEW_RADIANCE>())
152  {
153  if(metadata->getSurface().getEmitter())
154  {
155  SpectralStrength viewRadiance;
156  metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewRadiance);
157  addViewRadiance(pathThroughput * viewRadiance);
158  }
159  }
160 
161  // TODO: better handling of glossy optics
162  if(optics->getAllPhenomena().hasAtLeastOne({ESurfacePhenomenon::DIFFUSE_REFLECTION}) ||
163  pathLength >= m_maxViewpointDepth)
164  {
165  if constexpr(Viewpoint::template has<EViewpointData::SURFACE_HIT>()) {
166  m_viewpoint->template set<EViewpointData::SURFACE_HIT>(surfaceHit);
167  }
168  if constexpr(Viewpoint::template has<EViewpointData::VIEW_THROUGHPUT>()) {
169  m_viewpoint->template set<EViewpointData::VIEW_THROUGHPUT>(pathThroughput);
170  }
171  if constexpr(Viewpoint::template has<EViewpointData::VIEW_DIR>()) {
172  m_viewpoint->template set<EViewpointData::VIEW_DIR>(surfaceHit.getIncidentRay().getDirection().mul(-1));
173  }
174 
175  m_isViewpointFound = true;
176 
177  return ViewPathTracingPolicy().kill();
178  }
179  else
180  {
181  return ViewPathTracingPolicy().
182  traceSinglePathFor(ALL_ELEMENTALS).
183  useRussianRoulette(true);
184  }
185 }

187 template<typename Viewpoint, typename Photon>
188 inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onCameraSampleEnd()
189 {
190  if(!m_isViewpointFound)
191  {
192  return;
193  }
194 
195  TSurfaceEventDispatcher<ESaPolicy::STRICT> surfaceEvent(m_scene);
196 
197  const SurfaceHit& surfaceHit = m_viewpoint->template get<EViewpointData::SURFACE_HIT>();
198  const Vector3R L = m_viewpoint->template get<EViewpointData::VIEW_DIR>();
199  const Vector3R Ng = surfaceHit.getGeometryNormal();
200  const Vector3R Ns = surfaceHit.getShadingNormal();
201  const real R = m_viewpoint->template get<EViewpointData::RADIUS>();
202 
203  m_photonCache.clear();
204  m_photonMap->findWithinRange(surfaceHit.getPosition(), R, m_photonCache);
205 
206  // FIXME: as a parameter
207  const real alpha = 2.0_r / 3.0_r;
208 
209  const real N = m_viewpoint->template get<EViewpointData::NUM_PHOTONS>();
210  const real M = static_cast<real>(m_photonCache.size());
211  const real newN = N + alpha * M;
212  const real newR = (N + M) != 0.0_r ? R * std::sqrt(newN / (N + M)) : R;
213 
214  SpectralStrength tauM(0);
215  BsdfEvaluation bsdfEval;
216  for(const auto& photon : m_photonCache)
217  {
218  const Vector3R V = photon.template get<EPhotonData::FROM_DIR>();
219 
220  bsdfEval.inputs.set(surfaceHit, L, V, ALL_ELEMENTALS, ETransport::IMPORTANCE);
221  if(!surfaceEvent.doBsdfEvaluation(surfaceHit, bsdfEval))
222  {
223  continue;
224  }
225 
226  SpectralStrength tau = photon.template get<EPhotonData::THROUGHPUT_RADIANCE>();
227  tau.mulLocal(bsdfEval.outputs.bsdf);
228  tau.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
229 
230  tauM.addLocal(tau);
231  }
232  tauM.mulLocal(m_viewpoint->template get<EViewpointData::VIEW_THROUGHPUT>());
233 
234  const SpectralStrength tauN = m_viewpoint->template get<EViewpointData::TAU>();
235  const SpectralStrength newTau = (N + M) != 0.0_r ? (tauN + tauM) * (newN / (N + M)) : SpectralStrength(0);
236 
237  m_viewpoint->template set<EViewpointData::RADIUS>(newR);
238  m_viewpoint->template set<EViewpointData::NUM_PHOTONS>(newN);
239  m_viewpoint->template set<EViewpointData::TAU>(newTau);
240 }

242 template<typename Viewpoint, typename Photon>
243 inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onSampleBatchFinished()
244 {
245  // evaluate radiance using current iteration's data
246  for(int64 y = m_filmRegion.minVertex.y; y < m_filmRegion.maxVertex.y; ++y)
247  {
248  for(int64 x = m_filmRegion.minVertex.x; x < m_filmRegion.maxVertex.x; ++x)
249  {
250  const std::size_t viewpointIdx =
251  (y - m_film->getEffectiveWindowPx().minVertex.y) * static_cast<std::size_t>(m_film->getEffectiveResPx().x) +
252  (x - m_film->getEffectiveWindowPx().minVertex.x);
253 
254  PH_ASSERT_LT(viewpointIdx, m_numViewpoints);
255  const auto& viewpoint = m_viewpoints[viewpointIdx];
256 
257  const real radius = viewpoint.template get<EViewpointData::RADIUS>();
258  const real kernelArea = radius * radius * constant::pi<real>;
259  const real radianceMultiplier = 1.0_r / (kernelArea * static_cast<real>(m_numPhotonPaths));
260 
261  SpectralStrength radiance(viewpoint.template get<EViewpointData::TAU>() * radianceMultiplier);
262  radiance.addLocal(viewpoint.template get<EViewpointData::VIEW_RADIANCE>() / static_cast<real>(m_numSamplesPerPixel));
263  m_film->setPixel(static_cast<float64>(x), static_cast<float64>(y), radiance);
264  }
265  }
