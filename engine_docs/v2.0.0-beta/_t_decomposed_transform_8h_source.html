175 template<typename T>
176 inline bool TDecomposedTransform<T>::isScaleUniform(const T margin) const
177 {
178  const T dSxSy = std::abs(m_scale.x - m_scale.y);
179  const T dSySz = std::abs(m_scale.y - m_scale.z);
180  const T dSzSx = std::abs(m_scale.z - m_scale.x);
181 
182  return dSxSy < margin && dSySz < margin && dSzSx < margin;
183 }
