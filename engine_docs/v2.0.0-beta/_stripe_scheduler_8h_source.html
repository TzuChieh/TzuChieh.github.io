52 inline void StripeScheduler::scheduleOne(WorkUnit* const out_workUnit)
53 {
54  PH_ASSERT(out_workUnit);
55 
56  if(m_numScheduled < m_numWorkers)
57  {
58  const auto sideRange = math::ith_evenly_divided_range(m_numScheduled, m_sideLength, m_numWorkers);
59 
60  Region stripRegion = m_totalWorkUnit.getRegion();
61  stripRegion.minVertex[m_slicedAxis] += static_cast<int64>(sideRange.first);
62  stripRegion.maxVertex[m_slicedAxis] -= static_cast<int64>(m_sideLength - sideRange.second);
63 
64  *out_workUnit = WorkUnit(stripRegion, m_totalWorkUnit.getDepth());
65 
66  ++m_numScheduled;
67  }
68  else
69  {
70  *out_workUnit = WorkUnit();
71  }
72 }
