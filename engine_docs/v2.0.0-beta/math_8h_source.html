238 inline bool solve_linear_system_2x2(
239  const std::array<std::array<T, 2>, 2>& A,
240  const std::array<T, 2>& b,
241  std::array<T, 2>* const out_x)
242 {
243  PH_ASSERT(!std::numeric_limits<T>::is_integer);
244 
245  const T determinant = A[0][0] * A[1][1] - A[1][0] * A[0][1];
246  if(std::abs(determinant) <= std::numeric_limits<T>::epsilon())
247  {
248  return false;
249  }
250 
251  const T reciDeterminant = 1 / determinant;
252 
253  (*out_x)[0] = (A[1][1] * b[0] - A[0][1] * b[1]) * reciDeterminant;
254  (*out_x)[1] = (A[0][0] * b[1] - A[1][0] * b[0]) * reciDeterminant;
255  return true;
