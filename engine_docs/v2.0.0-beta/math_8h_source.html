395 
396  // Note that arbitrary number of bits can be supported by careful handling of bitwise operations

435 template<std::unsigned_integral UIntType, std::integral RangeType>
436 inline UIntType set_bits_in_range(const UIntType bits, const RangeType beginBitIdx, const RangeType endBitIdx)
437 {
438  // Inclusive as empty range is allowed, e.g., `beginBitIdx` == `endBitIdx`
439  PH_ASSERT_IN_RANGE_INCLUSIVE(beginBitIdx, 0, endBitIdx);
440  PH_ASSERT_IN_RANGE_INCLUSIVE(endBitIdx, beginBitIdx, sizeof_in_bits<UIntType>());
441 
442  // Mask for the bits in range. Constructed by producing required number of 1's then shift
443  // by <beginBitIdx>.
444  // Note that if there is integer promotion to signed types, it should be fine--promoted type
445  // must be wider than the original unsigned type, the bit shifts will never reach sign bit.
446  //
447  const auto bitMask = static_cast<UIntType>(((UIntType(1) << (endBitIdx - beginBitIdx)) - 1) << beginBitIdx);
448 
449  return bits | bitMask;
450 }

491  constexpr T TOTAL_SIZE = MAX - MIN;
