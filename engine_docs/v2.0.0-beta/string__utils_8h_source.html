323 inline void throw_from_std_errc_if_has_error(const std::errc errorCode)
324 {
325  // According to several sources, 0, or zero-initialized std::errc,
326  // indicates no error.
327  //
328  // [1] see the example for std::from_chars
329  // https://en.cppreference.com/w/cpp/utility/from_chars
330  // [2] https://stackoverflow.com/a/63567008
331  //
332  constexpr std::errc NO_ERROR_VALUE = std::errc();
333 
334  switch(errorCode)
335  {
336  case NO_ERROR_VALUE:
337  return;
338 
339  case std::errc::invalid_argument:
340  throw std::invalid_argument("input cannot be interpreted as a numeric value");
341 
342  case std::errc::result_out_of_range:
343  throw std::overflow_error("result will overflow the arithmetic type");
344 
345  case std::errc::value_too_large:
346  throw std::out_of_range("result cannot fit in the output buffer");
347 
348  default:
349  throw std::runtime_error(
350  "unknown error: std::errc = " + std::to_string(
351  static_cast<std::underlying_type_t<std::errc>>(errorCode)));
352  }
353 }
