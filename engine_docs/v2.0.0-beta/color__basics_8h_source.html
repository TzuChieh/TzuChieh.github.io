139 template<typename T = ColorValue>
140 inline TTristimulusValues<T> CIEXYZ_of_custom_reference_white(const TChromaticityValues<T>& refWhite)
141 {
142  // Reference whites should not have any component being <= 0
143  PH_ASSERT_GT(refWhite[0], static_cast<T>(0));
144  PH_ASSERT_GT(refWhite[1], static_cast<T>(0));
145 
146  const T x = refWhite[0];
147  const T y = refWhite[1];
148 
149  // Y is set to 1
150  const T Y_over_y = static_cast<T>(1) / y;
151  return
152  {
153  Y_over_y * x,
154  1.0,
155  Y_over_y * (static_cast<T>(1) - x - y)
156  };
157 }
