14 // A class defining a piecewise linear function, where points (x, y) are

18 // its nearest point's y value. If the function have got no points,

52 template<typename T>
53 inline T TPiecewiseLinear1D<T>::evaluate(const T x) const
54 {
55  if(m_points.empty())
56  {
57  return 0;
58  }
59 
60  // handling out-of-domain situation
61  // (this also handles if there's only 1 unique point)
62  if (x <= m_points.front().x()) return m_points.front().y();
63  else if(x >= m_points.back().x()) return m_points.back().y();
64 
65  const auto& result = std::lower_bound(m_points.begin(), m_points.end(), TVector2<T>(x),
66  &TPiecewiseLinear1D::pointDomainComparator);
67  PH_ASSERT(result != m_points.begin() && result != m_points.end());
68  const std::size_t i1 = result - m_points.begin();
69  const std::size_t i0 = i1 - 1;
70 
71  return evaluate(x, i0, i1);
72 }

74 template<typename T>
75 inline T TPiecewiseLinear1D<T>::evaluate(const T x,
76  const std::size_t p0Index,
77  const std::size_t p1Index) const
78 {
79  PH_ASSERT(p0Index < m_points.size() && p1Index < m_points.size());
80 
81  const TVector2<T>& p0 = m_points[p0Index];
82  const TVector2<T>& p1 = m_points[p1Index];
83  return p0.x() != p1.x() ? (x - p0.x()) / (p1.x() - p0.x()) * (p1.y() - p0.y()) + p0.y()
84  : (p0.y() + p1.y()) / 2;
85 }
