15 // A class defining a piecewise linear function, where points (x, y) are
16 // connected by straight lines; function values between two nearby points
17 // are evaluated by linearly interpolating their y-coordinates. Function
18 // values beyond the domain of specified x coordinates are evaluated with
19 // its nearest point's y value. If the function have got no points,
20 // evaluation results in zero no matter the inputs.
21 //
22 // Notice that update() must be called before feeding this function into
23 // any other methods.

54 template<typename T>
55 inline T TPiecewiseLinear1D<T>::evaluate(const T x) const
56 {
57  if(m_points.empty())
58  {
59  return 0;
60  }
61 
62  // handling out-of-domain situation
63  // (this also handles if there's only 1 unique point)
64  if (x <= m_points.front().x) return m_points.front().y;
65  else if(x >= m_points.back().x) return m_points.back().y;
66 
67  const auto& result = std::lower_bound(m_points.begin(), m_points.end(), TVector2<T>(x),
68  &TPiecewiseLinear1D::pointDomainComparator);
69  PH_ASSERT(result != m_points.begin() && result != m_points.end());
70  const std::size_t i1 = result - m_points.begin();
71  const std::size_t i0 = i1 - 1;
72 
73  return evaluate(x, i0, i1);
74 }

76 template<typename T>
77 inline T TPiecewiseLinear1D<T>::evaluate(const T x,
78  const std::size_t p0Index,
79  const std::size_t p1Index) const
80 {
81  PH_ASSERT(p0Index < m_points.size() && p1Index < m_points.size());
82 
83  const TVector2<T>& p0 = m_points[p0Index];
84  const TVector2<T>& p1 = m_points[p1Index];
85  return p0.x != p1.x ? (x - p0.x) / (p1.x - p0.x) * (p1.y - p0.y) + p0.y
86  : (p0.y + p1.y) / 2;
87 }
