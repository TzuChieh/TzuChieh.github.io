43 template<typename Func, typename R, typename... Args>
44 concept CEmptyFunctorForm =
45  std::is_empty_v<std::decay_t<Func>> &&// so we do not need to store its states
46  std::is_default_constructible_v<std::decay_t<Func>> &&// we construct it on every call

49 template<typename Func, typename R, typename... Args>
50 concept CNonEmptyTrivialFunctorForm =
51  !std::is_empty_v<std::decay_t<Func>> &&// to disambiguate from the empty form
52  !std::is_function_v<std::remove_pointer_t<std::decay_t<Func>>> &&// to disambiguate from the free function form
53  std::is_constructible_v<std::decay_t<Func>, Func> &&// we placement new using existing instance
54  std::is_trivially_copyable_v<std::decay_t<Func>> &&// so copying the underlying buffer is legal
55  std::is_trivially_destructible_v<std::decay_t<Func>> &&// somewhat redundant as we have `is_trivially_copyable`, but nice to be explicit
56  std::is_invocable_r_v<R, const std::decay_t<Func>&, Args...>;// must be const as we store its states and `operator ()` is `const`
